  • 前缀和

    • 一维前缀和

      定义 si=∑j=1iajs_i=\sum_{j=1}^ia_jsi​=∑j=1i​aj​ ,等价于 si=ai+si−1(s0=0)s_i=a_i+s_{i-1} (s_0=0)si​=ai​+si−1​(s0​=0)

      作用：对于固定数组a{a}a快速求区间和

      ∑i=lrai=s[r]−s[l−1]\sum_{i=l}^ra_i=s[r]-s[l-1]∑i=lr​ai​=s[r]−s[l−1]

      代码

      int a[N],s[N];
for(int i=1;i<=n;i++){
  cin>>a[i];
  s[i]=s[i-1]+a[i];
}//O(n)预处理
int q;
cin>>q;
while(q--){
  int l,r;
  cin>>l>>r;
  int ans=s[r]-s[r-1];
  cout<<ans<<endl;
  }//q次O(1)询问

    • 二维前缀和

      定义

      s[n][m]=∑i=1n∑j=1ma[i][j]=s[n−1][m]+s[n][m−1]−s[n−1][m−1]+a[n][m]\begin{align*} s[n][m] &= \sum_{i=1}^n\sum_{j=1}^ma_[i][j] \\ &= s[n-1][m]+s[n][m-1]-s_[n-1][m-1]+a[n][m] \end{align*}s[n][m]​=i=1∑n​j=1∑m​a[​i][j]=s[n−1][m]+s[n][m−1]−s[​n−1][m−1]+a[n][m]​

      常见于矩阵，方阵求和：

      左上角为(x1,y1)(x_1,y_1)(x1​,y1​)右下角为(x2,y2)(x_2,y_2)(x2​,y2​)的矩阵中所有元素之和可表示为:
      ∑i=x1x2∑j=y1y2a[i][j]=s[x2][y2]−s[x1−1][y2]−s[x2][y1−1]+s[x1−1][y1−1]\begin{align*} \sum_{i=x_1}^{x_2}\sum_{j=y_1}^{y_2}a[i][j] &= s[x_2][y_2]-s[x_1-1][y_2]-s[x_2][y_1-1]+s[x_1-1][y_1-1] \end{align*}i=x1​∑x2​​j=y1​∑y2​​a[i][j]​=s[x2​][y2​]−s[x1​−1][y2​]−s[x2​][y1​−1]+s[x1​−1][y1​−1]​

      (容斥原理)

      代码

      int a[N][N],s[N][N];
for(int i=1;i<=n;i++){
  for(int j=1;j<=m;j++){
    cin>>a[i][j];
    s[i][j]=s[i][j-1]+a[i-1][j]-s[i-1][j-1]+a[i][j];
  }
}//O(nm)预处理
int q;

while(q--){

  cout<<ans<<endl;
}//q次O(1)询问

    • 二阶差分

      与一阶没有本质区别,运用容斥原理就出来了

      定义：b[i][j]=a[i][j]−a[i−1][j]−a[i][j−1]+a[i−1][j+1]b[i][j]=a[i][j]-a[i-1][j]-a[i][j-1]+a[i-1][j+1]b[i][j]=a[i][j]−a[i−1][j]−a[i][j−1]+a[i−1][j+1]

      int a[N][N],b[N][N];
int nm;
cin>>n>>m;

for(int i=1;i<=n;i++){
  for(int j=1;j<=m;j++){
    cin>>a[i][j];
    b[i][j]=a[i][j]-a[i-1][j]-a[i][j-1]+a[i-1][j-1];
  }
}
int q;
cin>>q;
while(q--){
  int f_x,f_y,s_x,s_y,data;
  cin>>f_x>>f_y>>s_x>>s_y>>data;
  b[f_x][f_y]+=data;
  b[f_x][s_y+1]-=data;
  b[s_x+1][f_y]-=data;
  b[s_x+1][s_y+1]+=data;
} // O(1)处理
for(int i=1;i=n;i++){
  for(int j=1;j<=m;j++){
    a[i][j]=b[i][j]+a[i-1][j]+a[i][j-1]-a[i-1][j-1];
  }
}// O(nm)还原
